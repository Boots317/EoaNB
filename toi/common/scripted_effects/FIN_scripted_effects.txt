# returns FIN_var_famine_level: The level of famine we are on.
FIN_get_famine_level = {
	FIN = {
		if = {
			limit = { check_variable = { FIN_var_famine < 11 } }

			set_variable = { FIN_var_famine_level = 0 }

			else_if = {
				limit = { check_variable = { FIN_var_famine < 21 } }

				set_variable = { FIN_var_famine_level = 1 }

				else_if = {
					limit = { check_variable = { FIN_var_famine < 41 } }

					set_variable = { FIN_var_famine_level = 2 }

					else_if = {
						limit = { check_variable = { FIN_var_famine < 61 } }

						set_variable = { FIN_var_famine_level = 3 }

						else_if = {
							limit = { check_variable = { FIN_var_famine < 81 } }

							set_variable = { FIN_var_famine_level = 4 }

							else_if = {
								limit = { check_variable = { FIN_var_famine < 101 } }

								set_variable = { FIN_var_famine_level = 5 }
							}
						}
					}
				}
			}
		}
	}
}

# Method to change FIN_var_famine safely.
# arg FIN_arg_change: How much to change FIN_var_famine, could be both plus and minus.
####################

# FIN_change_famine = yes

FIN_change_famine = {
	add_to_variable = { FIN_var_famine = FIN_arg_change }

	FIN_get_famine_level = yes
}
