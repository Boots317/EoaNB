# returns FIN_var_famine_level: The level of famine we are on.
FIN_get_famine_level = {
	FIN = {
		if = {
			limit = { FIN_var_famine < 11 }

			set_temp_variable = { FIN_var_famine_level = 0 }

			else_if = {
				limit = { FIN_var_famine < 21 }

				set_temp_variable = { FIN_var_famine_level = 1 }

				else_if = {
					limit = { FIN_var_famine < 41 }

					set_temp_variable = { FIN_var_famine_level = 2 }

					else_if = {
						limit = { FIN_var_famine < 61 }

						set_temp_variable = { FIN_var_famine_level = 3 }

						else_if = {
							limit = { FIN_var_famine < 81 }

							set_temp_variable = { FIN_var_famine_level = 4 }

							else_if = {
								limit = { FIN_var_famine < 101 }

								set_temp_variable = { FIN_var_famine_level = 5 }
							}
						}
					}
				}
			}
		}
	}
}
