#ALL: Treaty of Tianjin
country_event = {
	id = secondopiumwarnew.5
	title = secondopiumwarnew.5.t
	desc = secondopiumwarnew.5.d

	fire_only_once = yes
	is_triggered_only = yes

	option = {
		name = secondopiumwarnew.5.a
		ai_chance = { factor = 95  }
		QNG = {
			CHI_idea_tianjin_treaty
		}
		if = {
			limit = {
				has_global_flag = SOW_anglo_french_yes
			}
			ENG = {
				add_timed_idea = {
					idea = CHI_idea_tianjin_treaty_euro
					days = 365
				}
				add_political_power = 200
			}
			FRA = {
				add_timed_idea = {
					idea = CHI_idea_tianjin_treaty_euro
					days = 365
				}
				add_political_power = 100
			}
		}
		else_if = {
			limit = {
				has_global_flag = SOW_taku_forts
			}
			ENG = {
				add_timed_idea = {
					idea = CHI_idea_tianjin_treaty_euro
					days = 365
				}
				add_political_power = 300
			}
			FRA = {
				add_timed_idea = {
					idea = CHI_idea_tianjin_treaty_euro
					days = 365
				}
				add_political_power = 150
			}
			USA = {
				add_timed_idea = {
					idea = CHI_idea_tianjin_treaty_euro
					days = 365
				}
			}
			RUS = {
				add_timed_idea = {
					idea = CHI_idea_tianjin_treaty_euro
					days = 365
				}
			}
		}
		else = {
			ENG = {
				add_timed_idea = {
					idea = CHI_idea_tianjin_treaty_euro
					days = 365
				}
				add_political_power = 250
			}
		}
	}
}
