	desc = austriainternal.100.d
	picture = GFX_report_event_vienna_1
	
	is_triggered_only = yes

	option = {
		name = austriainternal.100.a
		ai_chance = { base = 100 }
		add_political_power = 50
		if = {
			limit = { 
				has_country_flag = debt_free_flag
			}
			if = {
				limit = {
					has_idea = egy_vic_debt 
				}
				remove_ideas = egy_vic_debt
			}
		}
		hidden_effect = {
			clr_country_flag = debt_free_flag
			remove_decision = AUS_decision_debt_repayments_done
			clear_variable = debt_level
		}
	}
}
